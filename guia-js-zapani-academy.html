Configurando o Ambiente no Windows

🗺 O que é este módulo?

Preparar o terreno antes de codar

Você vai instalar o VSCode (editor), o Node.js (runtime que roda JS no seu PC) e o plugin Code Runner — que executa seu arquivo com um único atalho, sem abrir terminal. Tudo pensado para Windows, sem complicação.

  • Instalar VSCode com um único clique
  • Instalar Node.js LTS (arquivo .msi simples)
  • Configurar Code Runner para rodar JS com Ctrl+Alt+N

1.1 — Instalando o VSCode

  1. 1
    Baixar

  2. 2
    Instalar

  3. 3
    Abrir

  1. 1

Estruturas Condicionais

🗺 O que é este módulo?

Tomar decisões no código

Condicionais permitem ao programa "pensar" — escolher caminhos diferentes com base em condições. É como um semáforo: se vermelho, pare; se verde, siga. Sem condicionais, todo código seria linear.

  • if/else if/else — múltiplas condições
  • switch — verificar um valor contra muitos casos
  • Ternário — if/else compacto em uma linha

5.1 — if / else if / else

javascript
const nota = 75;
if (nota >= 90)      console.log("⭐ Excelente!");
else if (nota >= 70) console.log("✅ Aprovado!");
else if (nota >= 50) console.log("⚠ Recuperação");
else                 console.log("❌ Reprovado");

5.2 — Switch

javascript
const dia = 3;
switch (dia) {
  case 1: console.log("Segunda"); break;
  case 3: console.log("Quarta");  break;
  case 5: console.log("Sexta");   break;
  default: console.log("Outro");
}

5.3 — Ternário

javascript
const idade = 20;
const acesso = idade >= 18 ? "✅ Permitido" : "❌ Negado";
console.log(acesso);

Funções são como "receitas" — escrita uma vez, usada quantas vezes precisar. São o coração da organização em programação. Sem elas, todo código seria uma bagunça de repetições.

Arrays & Objetos

🗺 O que é este módulo?

Coleções de dados — a espinha dorsal de todo app

Arrays são listas ordenadas e objetos são estruturas com propriedades nomeadas. Praticamente todo programa usa essas duas estruturas. Dominar map, filter e reduce é o que diferencia um dev JS iniciante de um intermediário.

  • map — transforma cada elemento
  • filter — filtra por condição
  • reduce — acumula em um único valor
  • Objetos e desestruturação

8.1 — Arrays e Métodos Modernos

javascript
const nums = [1,2,3,4,5];
const dobrados = nums.map(n => n*2);    // [2,4,6,8,10]
const pares    = nums.filter(n => n%2===0); // [2,4]
const soma     = nums.reduce((a,n) => a+n, 0); // 15
const primeiro = nums.find(n => n>3);      // 4

8.2 — Objetos e Desestruturação

javascript
const aluno = { nome: "Zapani", nota: 85, aprovado: true,
  resumo() { return `${this.nome} — ${this.nota}pts`; } };
const { nome, nota } = aluno;  // desestruturação
console.log(nome, nota);       // "Zapani" 85
console.log(aluno.resumo());  // "Zapani — 85pts"

Programação Assíncrona

🗺 O que é este módulo?

Fazer coisas sem travar o programa

JavaScript executa uma coisa por vez, mas operações como buscar dados de APIs levam tempo. Async/await permite "pausar" elegantemente a execução enquanto espera — como fazer um pedido no restaurante sem travar o garçom.

  • async/await — forma moderna de código assíncrono
  • Promise — representa uma operação futura
  • fetch — buscar dados de APIs externas
  • try/catch — tratar erros assíncronos

9.1 — async / await com fetch

javascript
async function buscarPost(id) {
  try {
    const res = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`);
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    const data = await res.json();

javascript — auth.js
// Simulação simples de sessão por token (sem JWT para manter didático)
const sessoes = new Map();

function criarToken(userId) {
  const token = Math.random().toString(36).slice(2);
  sessoes.set(token, userId);
  return token;
}

function autenticar(req, res, next) {
  const token = req.headers["x-token"];
  if (!token || !sessoes.has(token))
    return res.status(401).json({ erro: "Não autenticado. Faça login primeiro." });
  req.userId = sessoes.get(token);
  next();
}

module.exports = { criarToken, autenticar };

async function fazerLogin(){
  const email=document.getElementById('login-email').value.trim();
  const senha=document.getElementById('login-senha').value;
  if(!email||!senha)return setAuthMsg('Preencha email e senha.');
  try{
    const r=await fetch(API+'/login',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({email,senha})});
    const d=await r.json();
    if(!r.ok)return setAuthMsg(d.erro||'Credenciais inválidas.');
    TOKEN=d.token;USER_NAME=d.nome;
    localStorage.setItem('tm_token',TOKEN);localStorage.setItem('tm_name',USER_NAME);
    entrarNoApp(USER_NAME);
  }catch(e){setAuthMsg('❌ Servidor offline. Rode node server.js primeiro.');}
}

    const r=await fetch(API+'/registrar',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({nome,email,senha})});
    const d=await r.json();
    if(!r.ok)return setAuthMsg(d.erro||'Erro ao criar conta.');
    setAuthMsg('✅ Conta criada! Faça login.');switchTab('login');
    document.getElementById('login-email').value=email;
  }catch(e){setAuthMsg('❌ Servidor offline.');}
}

async function carregarTarefas(){
  try{
    const r=await fetch(API+'/tarefas',{headers:{'x-token':TOKEN}});
    if(r.status===401){sair();return;}

function renderTarefas(){
  const busca=document.getElementById('search-input').value.toLowerCase();
  let f=todasTarefas.filter(t=>{
    if(filtroAtual==='abertas'&&t.concluida)return false;
    if(filtroAtual==='concluidas'&&!t.concluida)return false;
    if(catAtual&&t.categoria!==catAtual)return false;
    if(priosAtivas.size&&!priosAtivas.has(t.prioridade))return false;
    if(busca&&!t.titulo.toLowerCase().includes(busca))return false;
    return true;
  });
  document.getElementById('tasks-count').textContent=f.length+' '+(f.length===1?'item':'itens');
  const lista=document.getElementById('task-list');
  if(!f.length){lista.innerHTML='<div class="empty-state"><div class="empty-icon">📭</div><div class="empty-title">Nenhuma tarefa aqui</div><div class="empty-sub">Clique em "+ Nova Tarefa" para começar.</div></div>';return;}
  const o={alta:0,media:1,baixa:2};
  f.sort((a,b)=>a.concluida-b.concluida||o[a.prioridade]-o[b.prioridade]);
  lista.innerHTML=f.map(t=>`
    <div class="task-card ${t.prioridade} ${t.concluida?'done':''}" id="tc-${t.id}">
      <button class="task-check ${t.concluida?'checked':''}" onclick="toggleConcluida(${t.id},${t.concluida?0:1})"></button>
      <div class="task-body">
        <div class="task-title">${esc(t.titulo)}</div>
        <div class="task-meta">
          <span class="task-cat"># ${esc(t.categoria||'geral')}</span>
          <span class="task-prio ${t.prioridade}">${t.prioridade}</span>
          <span class="task-date">${fmtData(t.criado)}</span>
        </div>
      </div>
      <div class="task-actions">
        <button class="task-action-btn edit" onclick="abrirEdicao(${t.id})">✏</button>
        <button class="task-action-btn del" onclick="deletarTarefa(${t.id})">✕</button>
      </div>
    </div>
  `).join('');
}
async function toggleConcluida(id,v){
  await fetch(API+'/tarefas/'+id,{method:'PUT',headers:{'Content-Type':'application/json','x-token':TOKEN},body:JSON.stringify({concluida:v})});
  const t=todasTarefas.find(x=>x.id===id);if(t)t.concluida=v;
  renderTarefas();atualizarStats();toast(v?'✅ Concluída!':'↩ Reaberta','success');
}
async function deletarTarefa(id){
  const c=document.getElementById('tc-'+id);if(c)c.style.opacity='.3';
  await fetch(API+'/tarefas/'+id,{method:'DELETE',headers:{'x-token':TOKEN}});
  todasTarefas=todasTarefas.filter(t=>t.id!==id);
  renderTarefas();atualizarStats();renderCategorias();toast('🗑 Removida','success');
}

async function salvarTarefa(){
  const titulo=document.getElementById('m-titulo').value.trim();
  if(!titulo){document.getElementById('m-titulo').focus();return;}
  const body={titulo,descricao:document.getElementById('m-desc').value.trim(),prioridade:document.getElementById('m-prio').value,categoria:document.getElementById('m-cat').value.trim()||'geral'};
  const btn=document.getElementById('btn-save');btn.innerHTML='<span class="spinner"></span>';
  try{
    if(editandoId){
      const r=await fetch(API+'/tarefas/'+editandoId,{method:'PUT',headers:{'Content-Type':'application/json','x-token':TOKEN},body:JSON.stringify(body)});
      const updated=await r.json();
      const idx=todasTarefas.findIndex(t=>t.id===editandoId);if(idx!==-1)todasTarefas[idx]=updated;
      toast('✅ Atualizada!','success');
    }else{
      const r=await fetch(API+'/tarefas',{method:'POST',headers:{'Content-Type':'application/json','x-token':TOKEN},body:JSON.stringify(body)});
      const nova=await r.json();
      todasTarefas.unshift({...nova,concluida:0,criado:new Date().toISOString()});
      toast('✅ Criada!','success');
    }
    fecharModal();renderTarefas();atualizarStats();renderCategorias();
  }catch(e){toast('Erro ao salvar','error');}
  btn.textContent=editandoId?'Salvar Alterações':'Criar Tarefa';
}

function esc(s){return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');}
function fmtData(d){
  if(!d)return'';
  const dt=new Date(d.replace(' ','T')),agora=new Date(),diff=Math.floor((agora-dt)/1000);
  if(diff<60)return'agora';if(diff<3600)return Math.floor(diff/60)+'min';
  if(diff<86400)return Math.floor(diff/3600)+'h';
  return dt.toLocaleDateString('pt-BR',{day:'2-digit',month:'2-digit'});
}

}
